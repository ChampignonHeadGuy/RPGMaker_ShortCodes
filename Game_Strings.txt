#==============================================================================
# ** Game_Strings
#------------------------------------------------------------------------------
#  This class handles strings. It's a wrapper for the built-in class "Array."
#  Refer to "$strings" for the instance of this class.
#==============================================================================

class Game_Strings
  #--------------------------------------------------------------------------
  # * Object Initialization
  #--------------------------------------------------------------------------
  def initialize
    @data = []
  end
  #--------------------------------------------------------------------------
  # * Get String
  #     string_id : string ID
  #--------------------------------------------------------------------------
  def [](string_id)
    if string_id.between?(0, 1000) and @data[string_id] != nil
      return @data[string_id]
    else
      return ""
    end
  end
  #--------------------------------------------------------------------------
  # * Set String
  #     string_id : string ID
  #     text      : the string's text
  #--------------------------------------------------------------------------
  def []=(string_id, text)
    if string_id.between?(0, 1000) and text.is_a?(String)
      @data[string_id] = text
    end
  end
end

#==============================================================================
# ** Scene_Title
#------------------------------------------------------------------------------
#  This class performs title screen processing.
#==============================================================================

class Scene_Title
  #--------------------------------------------------------------------------
  # * Object Initialization
  #--------------------------------------------------------------------------
  def initialize
    $strings = Game_Strings.new
  end
end
