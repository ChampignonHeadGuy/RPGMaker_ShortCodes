#==============================================================================
# ** Game_Strings
#------------------------------------------------------------------------------
#  This class handles strings. It's a wrapper for the built-in class "Array."
#  Refer to "$strings" for the instance of this class.
#==============================================================================

class Game_Strings
  #--------------------------------------------------------------------------
  # * Object Initialization
  #--------------------------------------------------------------------------
  def initialize
    @data = []
  end
  #--------------------------------------------------------------------------
  # * Get String
  #     string_id : string ID
  #--------------------------------------------------------------------------
  def [](string_id)
    if string_id.between?(0, 1000) and @data[string_id] != nil
      return @data[string_id]
    else
      return ""
    end
  end
  #--------------------------------------------------------------------------
  # * Set String
  #     string_id : string ID
  #     text      : the string's text
  #--------------------------------------------------------------------------
  def []=(string_id, text)
    if string_id.between?(0, 1000) and text.is_a?(String)
      @data[string_id] = text
    end
  end
  #--------------------------------------------------------------------------
  # * Clear
  #--------------------------------------------------------------------------
  def clear
    @data.clear
  end
  #--------------------------------------------------------------------------
  # * Get Size
  #--------------------------------------------------------------------------
  def size
    return @data.size
  end
  #--------------------------------------------------------------------------
  # * Get Length
  #--------------------------------------------------------------------------
  def length
    return @data.length
  end
  #--------------------------------------------------------------------------
  # * Remove Last String
  #--------------------------------------------------------------------------
  def pop
    @data.pop
  end
  #--------------------------------------------------------------------------
  # * Get Reversed Data
  #--------------------------------------------------------------------------
  def reverse
    return @data.reverse
  end
  #--------------------------------------------------------------------------
  # * Set Reversed Data
  #--------------------------------------------------------------------------
  def reverse!
    @data.reverse!
  end
end

#==============================================================================
# ** Scene_Title
#------------------------------------------------------------------------------
#  This class performs title screen processing.
#==============================================================================

class Scene_Title
  #--------------------------------------------------------------------------
  # * Object Initialization
  #--------------------------------------------------------------------------
  def initialize
    if $strings == nil
      $strings = Game_Strings.new
    end
  end
end
